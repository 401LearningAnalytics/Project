peer_assignment_submission_schema_parts

Description

The questions that a submitter must answer in their submission

Columns

NameDescription
peer_assignment_idAssignment that this submission schema part lives in
peer_assignment_submission_schema_part_idThis part's id within the assignment
peer_assignment_submission_schema_part_typeThe type ("fileUpload", "plainText", "richText", "url")
peer_assignment_submission_schema_part_orderThe order within the assignment (lowest comes first)
peer_assignment_submission_schema_part_promptThe prompt (a question for the submitter to answer)

SQL create statement

CREATE TABLE peer_assignment_submission_schema_parts (
    peer_assignment_id VARCHAR(50) 
    ,peer_assignment_submission_schema_part_id VARCHAR(50) 
    ,peer_assignment_submission_schema_part_type VARCHAR(50) 
    ,peer_assignment_submission_schema_part_order INT4 
    ,peer_assignment_submission_schema_part_prompt VARCHAR(65535) 
    ,PRIMARY KEY (peer_assignment_id, peer_assignment_submission_schema_part_id)
    ,FOREIGN KEY (peer_assignment_id) REFERENCES peer_assignments(peer_assignment_id)
);
