feedback_course_ratings

Description

Contains data for course ratings

Columns

NameDescription
course_idCourse id for the course receiving ratings
ualberta_feedback_user_idEncrypted Coursera user id for ualberta feedback data.
feedback_systemFeedback system: STAR for course ratings
feedback_ratingValue of the rating. Should be 1-5 for active course ratings
feedback_max_ratingMaximum value of the rating: 5 for course ratings
feedback_tsTimestamp when the feedback was left

SQL create statement

CREATE TABLE feedback_course_ratings (
    course_id varchar 
    ,ualberta_feedback_user_id VARCHAR(50) NOT NULL
    ,feedback_system varchar 
    ,feedback_rating int4 
    ,feedback_max_rating int4 
    ,feedback_ts timestamp 
    ,PRIMARY KEY (course_id, feedback_system, ualberta_feedback_user_id, feedback_ts)
    ,FOREIGN KEY (course_id) REFERENCES courses(course_id)
);
