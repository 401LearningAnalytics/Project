assessment_response_options

Description

learner responses to assessments

Columns

NameDescription
assessment_response_idUnique id for each of users' responses to an assessment.
assessment_option_idunique choice option id, linking to the assessment_option_id in the assessment_options table
assessment_response_correctWhether the response was correct
assessment_response_feedbackFeedback shown to the user on interacting with the assessment.
assessment_response_selectedWhether this response option was selected

SQL create statement

CREATE TABLE assessment_response_options (
    assessment_response_id VARCHAR(50) 
    ,assessment_option_id VARCHAR(50) 
    ,assessment_response_correct BOOL 
    ,assessment_response_feedback VARCHAR(20000) 
    ,assessment_response_selected BOOL 
    ,PRIMARY KEY (assessment_response_id, assessment_option_id)
    ,FOREIGN KEY (assessment_response_id) REFERENCES assessment_responses(assessment_response_id)
);
