peer_submission_part_free_responses

Description

Answers to the submission schema questions, for the "plainText" and "richText" submission schema part types.

Columns

NameDescription
peer_assignment_idThe assignment that this answer applies to
peer_assignment_submission_schema_part_idThe part of the submission schema that this answer applies to
peer_submission_idThe submission that this answer is part of
peer_submission_part_free_response_textThe contents of the answer

SQL create statement

CREATE TABLE peer_submission_part_free_responses (
    peer_assignment_id VARCHAR(50) 
    ,peer_assignment_submission_schema_part_id VARCHAR(50) 
    ,peer_submission_id VARCHAR(50) 
    ,peer_submission_part_free_response_text VARCHAR(32767) 
    ,PRIMARY KEY (peer_assignment_id, peer_assignment_submission_schema_part_id, peer_submission_id)
    ,FOREIGN KEY (peer_submission_id) REFERENCES peer_submissions(peer_submission_id)
    ,FOREIGN KEY (peer_assignment_id, peer_assignment_submission_schema_part_id) REFERENCES peer_assignment_submission_schema_parts(peer_assignment_id, peer_assignment_submission_schema_part_id)
    ,FOREIGN KEY (peer_assignment_id) REFERENCES peer_assignments(peer_assignment_id)
);
