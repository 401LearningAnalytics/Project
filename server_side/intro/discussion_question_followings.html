discussion_question_followings

Description

records when discussion question is followed or unfollowed

Columns

NameDescription
ualberta_discussions_user_idEncrypted Coursera user id for ualberta discussions data.
course_id22 character unique string identifying an individual course
discussion_question_idThe id of the discussion question
discussion_question_following_activetrue if following, false if not
discussion_question_following_tstimestamp of following status change

SQL create statement

CREATE TABLE discussion_question_followings (
    ualberta_discussions_user_id VARCHAR(50) NOT NULL
    ,course_id varchar 
    ,discussion_question_id varchar 
    ,discussion_question_following_active bool 
    ,discussion_question_following_ts timestamp 
    ,PRIMARY KEY (ualberta_discussions_user_id, discussion_question_id, discussion_question_following_ts)
);
